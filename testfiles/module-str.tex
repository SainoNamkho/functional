\BEGINTEST{Testing \StrConst}

\ENDTEST

\BEGINTEST{Testing \StrLog}

\ENDTEST

\ENDTEST

\ENDTEST

\ENDTEST

\ENDTEST

\ENDTEST

\BEGINTEST{Testing \StrIfEq}
\StrIfEqTF {abc} {abc} {\Result{Yes}} {\Result{No}}
\StrIfEqTF {abc} {xyz} {\Result{Yes}} {\Result{No}}
\StrSet \lTmpaStr {abc}
\StrSet \lTmpbStr {abc}
\StrSet \lTmpcStr {xyz}
\StrIfEqTF {\lTmpaStr} {\lTmpbStr} {\Result{Yes}} {\Result{No}}
\StrIfEqTF {\lTmpaStr} {\lTmpcStr} {\Result{Yes}} {\Result{No}}
\StrIfEqTF {\Value\lTmpaStr} {\Value\lTmpbStr} {\Result{Yes}} {\Result{No}}
\StrIfEqTF {\Value\lTmpaStr} {\Value\lTmpcStr} {\Result{Yes}} {\Result{No}}
\ENDTEST

\BEGINTEST{Testing \StrCompare}
\BoolVarLog {\StrCompare {ab} < {abc}}
\BoolVarLog {\StrCompare {ab} < {aa}}
\StrCompareTF {ab} < {abc} {\Result{Yes}} {\Result{No}}
\StrCompareTF {ab} < {aa} {\Result{Yes}} {\Result{No}}
\BoolVarLog {\StrIfCompare {ab} < {abc}}
\BoolVarLog {\StrIfCompare {ab} < {aa}}
\StrIfCompareTF {ab} < {abc} {\Result{Yes}} {\Result{No}}
\StrIfCompareTF {ab} < {aa} {\Result{Yes}} {\Result{No}}
\ENDTEST

\ENDTEST

}{

}{
  \Result {No ~ Match!}
}
\IgnoreSpacesOff
\ENDTEST
