\documentclass{article}

\input{regression-test}
\AtBeginDocument{\START}
%\AtEndDocument{\END}

\usepackage{functional}
\Functional{tracing=true}

\begin{document}

\BEGINTEST{Testing \BoolConst}
\BoolConst \cFooSomeBool {\IntCompare {3} > {2}}
\BoolVarLog \cFooSomeBool
\ENDTEST

\BEGINTEST{Testing \BoolSet}
\BoolSet \lTmpaBool {\IntCompare{3}<{4}}
\BoolVarLog \lTmpaBool
\BoolSet \lTmpaBool {\IntCompare{3}<{4} && \StrIfEq{abc}{uvw}}
\BoolVarLog \lTmpaBool
\ENDTEST

\BEGINTEST{Testing \BoolSetTrue}
\BoolSetTrue \lTmpaBool
\BoolVarIfTF \lTmpaBool {\Return{True!}} {\Return{False!}}
\ENDTEST

\BEGINTEST{Testing \BoolSetFalse}
\BoolSetFalse \lTmpaBool
\BoolVarIfTF \lTmpaBool {\Return{True!}} {\Return{False!}}
\ENDTEST

\BEGINTEST{Testing \BoolSetEq}
\BoolSetTrue \lTmpaBool
\BoolSetEq \lTmpbBool \lTmpaBool
\BoolVarLog \lTmpbBool
\ENDTEST

\BEGINTEST{Testing \BoolLog}
\BoolLog \lTmpaBool
\ENDTEST

\BEGINTEST{Testing \BoolIfExist}
\BoolIfExistTF \lTmpaBool {\Return{Yes}} {\Return{No}}
\BoolIfExistTF \lFooUndefinedBool {\Return{Yes}} {\Return{No}}
\ENDTEST

\BEGINTEST{Testing \BoolVarIf}
\BoolSetTrue\lTmpaBool
\BoolVarIfTF\lTmpaBool{\Return{True!}}{\Return{False!}}
\BoolSetFalse\lTmpaBool
\BoolVarIfTF\lTmpaBool{\Return{True!}}{\Return{False!}}
\ENDTEST

\BEGINTEST{Testing \BoolVarNot}
\BoolVarNotTF {\IntCompare {3} > {2}} {\Return{Yes}} {\Return{No}}
\ENDTEST

\BEGINTEST{Testing \BoolVarAnd}
\BoolVarAndTF {\IntCompare {3} > {2}} {\IntCompare {3} > {4}}
  {\Return{Yes}} {\Return{No}}
\ENDTEST

\BEGINTEST{Testing \BoolVarOr}
\BoolVarOrTF {\IntCompare {3} > {2}} {\IntCompare {3} > {4}}
  {\Return{Yes}} {\Return{No}}
\ENDTEST

\BEGINTEST{Testing \BoolVarXor}
\BoolVarXorTF {\IntCompare {3} > {2}} {\IntCompare {3} > {4}}
  {\Return{Yes}} {\Return{No}}
\ENDTEST

\end{document}
